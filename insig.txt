'use strict';
const {
  Model
} = require('sequelize');
module.exports = (sequelize, DataTypes) => {
  class SafRequestRejection  extends Model {
    static associate(models) {
      SafRequestRejection.hasMany(models.SafRequest, {
        foreignKey: 'rejectionId',
        as: 'requests'
      });
    }

    toJSON() {
      const { createdAt, updatedAt, ...data } = this.get();
      return data;
    }
  }
  SafRequestRejection.init({
    name: DataTypes.STRING
  }, {
    sequelize,
    modelName: 'SafRequestRejection',
    tableName: 'saf_request_rejections'
  });
  return SafRequestRejection;
};
