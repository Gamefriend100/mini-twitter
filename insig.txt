const db = require("../../../database/models");

const deleteBranchSupply = async (req, res) => {
  try {

    const { id } = req.params;

    const deleted = await db.SafKit.destroy({
      where: { id }
    });

    if (!deleted) {
      return res.status(404).json({
        msg: "No existe ese registro en el botiquín"
      });
    }

    return res.status(200).json({
      msg: "Insumo eliminado correctamente del botiquín"
    });

  } catch (error) {

    console.error(error);

    return res.status(500).json({
      msg: "Error al eliminar el insumo"
    });

  }
};

module.exports = {
  deleteBranchSupply
};
