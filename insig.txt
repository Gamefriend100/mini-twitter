import { UseRequestContext } from "../../../../context/safety/kit/request/RequestContext";
import { Modal } from "../../../../components/shared";
import { Routes, Route } from "react-router-dom";
import Request from './Request';
import Create from "./Create";
import Details from "./Details";
import Rejection from "./Rejection";

function Index() {
  const { modalRequest, setModalRequest } = UseRequestContext();

  return (
    <>
      <div>
        <Routes>
          <Route path="/" element={<Request />} />
          <Route path="/crear" element={<Create />} />
          <Route path="/detalles" element={<Details />}/>
          <Route path="/rechazar" element={<Rejection />} />
        </Routes>
      </div>
      <Modal infoModal={modalRequest} setInfoModal={setModalRequest} />
    </>
  );
}

export default Index;
