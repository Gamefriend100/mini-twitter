import { useQuery } from "@tanstack/react-query";
import { fetchData } from "../../../accions";

export const useGetSupplies = (token) => {
  const supplies = useQuery({
    queryKey: ["supplies"],
    queryFn: () => fetchData("supplies", token),
    refetchOnWindowFocus: false,
    enabled: !!token,
  });
  return {
    supplies,
  };
};
