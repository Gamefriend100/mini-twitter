const db = require('../../database/models');

const getRequestRejection = async (req, res) => {
    try {
        const rejections = await db.SafRequestRejection.findAll({
            attributes: [["id", "value"],
            ["name", "label"]]
        });

        return res.json(rejections);
    } catch (error) {
        return res.status(500).json({ msg: error.message });
    }
};

module.exports = {
    getRequestRejection
};
