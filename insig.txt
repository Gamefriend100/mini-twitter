const updateRequest = async (req, res) => {
    const { id } = req.params;
    const { supplies } = req.body;

    try {
        const request = await db.SafRequest.findByPk(id);

        if (!request) {
            return res.status(404).json({
                msg: "Solicitud no encontrada"
            });
        }

        const currentSupplies = await db.SafRequestSupply.findAll({
            where: { requestId: id }
        });

        const currentMap = new Map(
            currentSupplies.map(s => [s.supplyId, s])
        );

        for (const supplyId of supplies) {

            const existing = currentMap.get(supplyId);

            if (!existing) {
                
                await db.SafRequestSupply.create({
                    requestId: id,
                    supplyId
                });
            }
        }
        await request.reload({
            include: [
                {
                    model: db.User,
                    as: 'user',
                    attributes: ["associateNumber", "fullName"]
                },
                {
                    model: db.Branch,
                    as: 'branch',
                    attributes: ["id", "name"]
                },
                {
                    model: db.SafRequestStatus,
                    as: 'status',
                    attributes: ["id", "name"]
                },
                {
                    model: db.SafRequestRejection,
                    as: 'rejection',
                    attributes: ["id", "name"]
                },
                {
                    model: db.SafRequestSupply,
                    as: 'requestSupplies',
                    include: [
                        {
                            model: db.SafSupply,
                            as: 'supply',
                            attributes: [
                                ["id", "value"],
                                ["name", "label"]
                            ]
                        }
                    ]
                }
            ]
        });

        return res.status(200).json({
            msg: "Solicitud actualizada correctamente",
            updatedRequest: request
        });

    } catch (error) {
        return res.status(500).json({
            msg: error.message
        });
    }
};
