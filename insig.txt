import { Button, TitleCard } from "../../../../components/shared";
import { AddCircle, FactCheck } from "@mui/icons-material";
import { useNavigate } from "react-router-dom";
import { UseAppContext } from "../../../../context/AppContext";
import Table from "./Table";

const Index = () => {
  const goTo = useNavigate();
  const { authUser } = UseAppContext();
  console.log("authUser", authUser);

  const role = authUser?.user?.apps?.[0]?.role;
  const description =
    role === "SUCURSAL"
      ? "Gestión de mis solicitudes"
      : "Gestión de solicitudes de botiquines por sucursal";

  return (
    <div className="space-y-6">
      <TitleCard
        title="SOLICITUDES"
        desc={description}
        icon={<FactCheck />}
      >
        {role !== "TALENTO HUMANO" && (
          <Button
            variant="contained"
            size="small"
            text={"Crear Solicitud"}
            startIcon={<AddCircle sx={{ fontSize: 16 }} />}
            onClick={() => goTo("/seguridad/botiquin/solicitudes/crear")}
          />
        )}
      </TitleCard>

      <Table />
    </div>
  );
};

export default Index;
