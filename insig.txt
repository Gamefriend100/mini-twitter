const changeRequestStatus = async (req, res) => {
    const { id } = req.params;
    const { statusId, rejectionId = null, commit = null } = req.body;

    try {

        const request = await db.SafRequest.findByPk(id);

        if (!request) {
            return res.status(404).json({
                msg: "Solicitud no encontrada"
            });
        }

        // 🔹 Si es rechazo
        if (statusId === 5) {
            request.rejectionId = rejectionId;
            request.rejectionReason = clearString(commit);
        }

        // 🔹 Si pasa a entregado (6), crear kits
        if (statusId === 6) {

            const supplies = await db.SafRequestSupply.findAll({
                where: { requestId: request.id }
            });

            const kits = supplies.map(s => ({
                supplyId: s.supplyId,
                branchId: request.branchId,
                quantity: 0
            }));

            await db.SafKit.bulkCreate(kits);
        }

        // 🔹 Actualizar estado directamente
        request.statusId = statusId;

        await request.save();

        await request.reload({
            include: [
                {
                    model: db.User,
                    as: 'user',
                    attributes: ["associateNumber", "fullName"]
                },
                {
                    model: db.Branch,
                    as: 'branch',
                    attributes: ["id", "name"]
                },
                {
                    model: db.SafRequestStatus,
                    as: 'status',
                    attributes: ["id", "name"]
                },
                {
                    model: db.SafRequestRejection,
                    as: 'rejection',
                    attributes: ["id", "name"]
                }
            ]
        });

        return res.status(200).json({
            msg: "Solicitud actualizada correctamente",
            updatedRequest: request
        });

    } catch (error) {
        return res.status(500).json({
            msg: error.message
        });
    }
};
