import {
  TitleCard,
  Card,
  MUIAutocomplete,
  Button,
  TextFieldError,
  notifySuccess,
  notifyError,
  Check
} from '../../../../components/shared';

import { DirectionsRun, TaskAlt } from '@mui/icons-material';
import { useState } from 'react';
import { UseInspectionContext } from '../../../../context/safety/inspections/InspectionsContext';
import { UseAppContext } from '../../../../context/AppContext';
import { appApi } from '../../../../api';
import SelectedItem from './SelectedItem';
import { UseWalkthroughContext } from '../../../../context/safety/inspections/walkthrough/WalkthroughContext';

const Create = () => {
  const url = "/seguridad/inspecciones/recorridos";
  const { authUserToken, branches } = UseAppContext();
  const { evaluations } = UseInspectionContext();
  const [includeCorporate, setIncludeCorporate] = useState(false);
  const [isGeneral, setIsGeneral] = useState(false);

  const {
    addRemoveBranch,
    addRemoveEvaluation,
    errors,
    formState,
    handleInputChange,
    handleSubmit,
    selectedBranches,
    selectedEvaluations,
    setBranchesByZone,
    setSelectedBranches,
    zones,
    toggleAllEvaluations
  } = UseWalkthroughContext();


  return (
    <div className='space-y-6'>
      <TitleCard
        title="CREAR UN RECORRIDO"
        desc="Captura los datos correspondientes para poder registrar un nuevo recorrido"
        icon={<DirectionsRun />}
        url={url}
      />

      <Card>
        <form className="p-6 space-y-6" onSubmit={handleSubmit}>
          <TextFieldError
            required
            autoFocus
            name="description"
            label="Descripción del recorrido"
            value={formState.description}
            onChange={handleInputChange}
            errors={errors}
          />

          <TextFieldError
            required
            type="date"
            name="dueDate"
            label="Fecha límite para realizar el recorrido"
            shrink
            value={formState.dueDate}
            onChange={handleInputChange}
            errors={errors}
          />

          <div className="flex justify-between items-center mb-2">
            <p className="text-sm text-gray-600">TODAS LAS NORMAS</p>

            <input
              type="checkbox"
              checked={selectedEvaluations.length === evaluations.length}
              onChange={toggleAllEvaluations}
            />
          </div>

          <div>
            <p className='text-sm text-gray-600 mb-2'>Selecciona las evaluaciones que se aplicarán en este recorrido:</p>
            <div className='border border-gray-300 rounded-md p-4 flex gap-2 flex-wrap justify-center'>
              {
                evaluations.map(evaluation => (
                  <SelectedItem
                    key={evaluation.id}
                    item={evaluation}
                    options={selectedEvaluations}
                    addRemoveItem={addRemoveEvaluation}
                  />
                ))
              }
            </div>
          </div>

          <div>
            <div className='flex justify-between'>
              <p className='text-sm text-gray-600 mb-2'>Selecciona las sucursales donde se llevará a cabo el recorrido:</p>
              <div className='flex gap-2'>
                {
                  zones.map(zone => (
                    <div
                      key={zone.id}
                      className="flex items-center gap-2 text-sm"
                    >
                      <input
                        type='checkbox'
                        checked={zone.checked}
                        onChange={() => setBranchesByZone(zone.id)}
                      />
                      {zone.name}
                    </div>
                  ))
                }
              </div>
            </div>
            <div className='border border-gray-300 rounded-md p-4 flex gap-2 flex-wrap justify-center'>
              {
                branches.map(branch => (
                  <SelectedItem
                    key={branch.value}
                    item={branch}
                    options={selectedBranches}
                    fieldId={"value"}
                    fieldName={"label"}
                    addRemoveItem={addRemoveBranch}
                  />
                ))
              }
            </div>
          </div>

          <div className='flex justify-between items-center text-sm text-gray-600'>
            <p>¿El recorrido incluye el Corporativo?</p>

            <div className="flex items-center gap-2">
              <span
                className={`font-semibold ${includeCorporate ? "text-green-600" : "text-red-600"
                  }`}
              >
                {includeCorporate ? "SI" : "NO"}
              </span>

              <Check
                checked={includeCorporate}
                onAction={() => setIncludeCorporate(true)}
                offAction={() => setIncludeCorporate(false)}
              />
            </div>
          </div>

          {
            includeCorporate && (
              <div className='flex justify-between items-center text-sm mt-2 text-gray-600'>
                <p>¿El recorrido será por departamentos?</p>

                <div className="flex items-center gap-2">
                  <span
                    className={`font-semibold ${isGeneral ? "text-green-600" : "text-red-600"
                      }`}
                  >
                    {isGeneral ? "SI" : "NO"}
                  </span>

                  <Check
                    checked={isGeneral}
                    onAction={() => setIsGeneral(true)}
                    offAction={() => setIsGeneral(false)}
                  />
                </div>
              </div>
            )}

          <div className="flex justify-end border-t pt-4">
            <Button
              text="Guardar"
              icon={<TaskAlt fontSize="small" />}
              type="submit"
            />
          </div>
        </form>
      </Card>
    </div>
  );
};

export default Create;
