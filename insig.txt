import { createContext, useContext, useState, useEffect } from "react";
import { UseAppContext } from "../../../AppContext";
import { useGetKit } from "../../../../hooks/safety/kit";
import { appApi } from "../../../../api";

const BranchesContext = createContext();

export const UseBranchesContext = () => {
  return useContext(BranchesContext);
};

export const BranchesProvider = ({ children }) => {

  const [modalRequest, setModalRequest] = useState({ show: false });
  const [selectedKitId, setSelectedKitId] = useState(null);
  const [kitData, setKitData] = useState([]);

  const { authUserToken, branches, authUser, hasRole } = UseAppContext();
  const isBranchRole = hasRole(["SUCURSAL"]);

  const [selectBranch, setSelectBranch] = useState(null);

  useEffect(() => {

    if (isBranchRole && authUser?.user?.branchId) {
      setSelectBranch(authUser.user.branchId);

    } else if (branches?.length) {
      setSelectBranch(branches[0].value);
    }

  }, [branches, authUser, isBranchRole]);

  const { data, isFetching } = useGetKit(selectBranch, authUserToken).kit;

  useEffect(() => {
    if (data) {
      setKitData(data);
    }
  }, [data]);

  const deleteBranchSupply = async (id) => {
    try {

      await appApi.delete(`/intranet/branch/supplies/${id}`, {
        headers: { Authorization: `Bearer ${authUserToken}` }
      });

      setKitData(prev => prev.filter(item => item.id !== id));

      setModalRequest({ show: false });

    } catch (error) {
      console.error(error);
    }
  };

  return (
    <BranchesContext.Provider
      value={{
        selectBranch,
        setSelectBranch: isBranchRole ? () => {} : setSelectBranch,
        data: kitData,
        isFetching,
        modalRequest,
        setModalRequest,
        selectedKitId,
        setSelectedKitId,
        deleteBranchSupply
      }}
    >
      {children}
    </BranchesContext.Provider>
  );
};
