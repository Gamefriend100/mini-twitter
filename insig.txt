import { useGetRequest, useGetRequestRejection, useGetRequestStatus, useGetSuppliesByRequestId } from "../../../../hooks/safety/kit";
import { createContext, useContext, useEffect, useState } from "react";
import { UseAppContext } from "../../../AppContext";

const RequestContext = createContext();

export const UseRequestContext = () => {
  return useContext(RequestContext);
};

export const RequestProvider = ({ children }) => {
  const { authUserToken } = UseAppContext();
  const [modalRequest, setModalRequest] = useState({ show: false });
  const [selectedSupplies, setSelectedSupplies] = useState([]);
  const [requestList, setRequestList] = useState([]);
  const [selectedRequest, setSelectedRequest] = useState(null);

  const {
    data: initRequestStatus,
    isFetching: isFetchingRequestStatus,
    isSuccess: isSucessRequestStatus,
  } = useGetRequestStatus(authUserToken).requestStatus;


  const {
    data: initRequestRejection,
    isFetching: isFetchingRequestRejection,
    isSuccess: isSuccessRequestRejection
  } = useGetRequestRejection(authUserToken).requestRejection;

  const {
    data: initSuppliesByRequestId,
    isFetching: isFetchingSuppliesByRequestId,
    isSuccess: isSuccessSuppliesByRequestId
  } = useGetSuppliesByRequestId(selectedRequest?.id, authUserToken).requestSuppliesId;

  const {
    data: initRequestList,
    isFetching: isFetchingRequestList,
    isSuccess: isSuccessRequestList
  } = useGetRequest(authUserToken).request;

  const addSupply = (supplyId) => {
    setSelectedSupplies(prev => [...prev, supplyId]);
  };

  useEffect(() => {
    if (!isFetchingRequestList && isSuccessRequestList) {
      setRequestList(initRequestList)
    }
  }, [isFetchingRequestList]);

  return (
    <RequestContext.Provider
      value={{
        modalRequest,
        setModalRequest,
        selectedSupplies,
        setSelectedSupplies,
        addSupply,
        requestList,
        isFetchingRequestList,
        setRequestList,
        selectedRequest,
        setSelectedRequest,
        initSuppliesByRequestId,
        isFetchingSuppliesByRequestId,
        isSuccessSuppliesByRequestId,
        initRequestRejection,
        isFetchingRequestRejection,
        isSuccessRequestRejection,
        initRequestStatus
      }}
    >
      {children}
    </RequestContext.Provider>
  );
};
