import { TitleCard, Card, SelectBranch, Loader, Empty, Modal } from '../../../../components/shared'
import { Vaccines } from '@mui/icons-material'
import { UseBranchesContext } from '../../../../context/safety/kit/branches/BranchesContext';
import ItemSupply from '../../../../components/safety/kit/branches/ItemSupply';
import { UseAppContext } from '../../../../context/AppContext';

const Index = () => {

  const {
    data,
    isFetching,
    setSelectBranch,
    selectBranch,
    modalRequest,
    setModalRequest
  } = UseBranchesContext();
  

  const { hasRole } = UseAppContext();

  const title = hasRole(["SUCURSAL"])
    ? "BIENVENIDO A TU BOTIQUÍN"
    : "BOTIQUÍN POR SUCURSAL";

  return (
    <>
      <div className='space-y-6'>

        <TitleCard
          title={title}
          desc="Gestión de insumos"
          icon={<Vaccines />}
        />

        {hasRole(["ADMIN", "TALENTO HUMANO"]) && (
          <Card>
            <SelectBranch
              value={selectBranch}
              handleChange={({ value }) => setSelectBranch(value)}
            />
          </Card>
        )}

        <Card classes="flex flex-col gap-3">
          {isFetching
            ? (<Loader />)
            : data?.length === 0
              ? <Empty msg='No hay insumos asignados a esta sucursal' />
              : data?.map((element) => (
                <ItemSupply key={element.id} supply={element} />
              ))
          }
        </Card>

      </div>

      <Modal
        infoModal={modalRequest}
        setInfoModal={setModalRequest}
      />
    </>
  );
};

export default Index;
